docker build -t <название образа>:latest .- создание образа.
//ID образа можно посмотреть с помощью команды docker images
docker run --rm -d --name <название контейнера> –p 5432:5432 <ID образа> - запуск контейнера на основе образа.
docker exec -it <название контейнера> psql -U <имя пользователя> - запуск интерфейса psql.
cd /docker-entrypoint-initdb.d - команда, с помощью которой заходим в папку, где находится скрипт.
ls - просмотр содержимого.
сat init.sql - запись в файл (изменение файла) 
docker run --rm --name <название контейнера> -p 5432:5432 -v /data:/var/lib/postgresql/data -d <ID образа>
0f8bf63d8ffba0546ee195c0902ffaaf4a817274163860a4364ce417178cb474 - запуск контейнера с томом для хранения и управления данными.
